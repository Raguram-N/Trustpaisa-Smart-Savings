// Compare logic

function runCompare(){
const price=Number(document.getElementById('cmp-price').value)||0
const budget=Number(document.getElementById('cmp-budget').value)||0
const loan = Math.max(0, price - budget)
// simple interest assumptions for demo
const emiInterest = Math.round(loan * 0.333) // e.g., 33% total interest
const tpInterest = Math.round(loan * 0.166) // 16.6% total interest
const emiTotal = price - budget + emiInterest + budget
const tpTotal = price - budget + tpInterest + budget
document.getElementById('emi-interest').textContent = 'â‚¹' + emiInterest.toLocaleString()
document.getElementById('tp-interest').textContent = 'â‚¹' + tpInterest.toLocaleString()
const emiPct = Math.min(100, (emiInterest/(loan||1))*50 + 30)
const tpPct = Math.min(100, (tpInterest/(loan||1))*50 + 10)
document.getElementById('emi-bar').style.width = emiPct + '%'
document.getElementById('tp-bar').style.width = tpPct + '%'
document.getElementById('cmp-total').textContent = `Traditional EMI: â‚¹${emiTotal.toLocaleString()} â€” TrustPaisa: â‚¹${tpTotal.toLocaleString()}`
document.getElementById('cmp-result').style.display = 'block'
}

}
